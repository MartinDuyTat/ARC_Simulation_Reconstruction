11  double PhotonMirrorDistance(const Photon &photon, const RadiatorCell &radiatorCell) {
12  auto MirrorCentre = radiatorCell.GetMirrorCentre();
13  const double R = radiatorCell.GetMirrorCurvature();
14  const double a = 1.0;
15  const double b = -2*photon.m_Direction.Dot(MirrorCentre - photon.m_Position);
16  const double c = (MirrorCentre - photon.m_Position).Mag2() - R*R;
17  const double s1 = (-b + TMath::Sqrt(b*b - 4.0*a*c))/(2.0*a);
18  const double s2 = (-b - TMath::Sqrt(b*b - 4.0*a*c))/(2.0*a);
19  return std::max(s1, s2);
20  }

22  void TracePhotonToMirror(Photon &photon, const RadiatorCell &radiatorCell) {
